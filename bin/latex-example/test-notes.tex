\lecture{1}{May 23th, 2020 15:00}{One Dimensional Motion}

\begin{definition}
    Average Velocity $= \frac{\Delta x}{\Delta t}$
    Average Speed $= \frac{d}{\Delta t}$ \\
    Displacement is the area under $v$ vs $t$
\end{definition}
\begin{definition}
    Kinematic Equations \\
    Missing $a$: $\Delta x = \frac{1}{2}\left(v_0+v\right)t$ \\
    Missing $x$: $v=v_0 + at$ \\
    Missing $v$: $x = x_0 + v_0t + \frac{1}{2}at^2$ \\
    Missing $v_0$: $x = x_0 + vt - \frac{1}{2}at^2$ \\
    Missing $t$: $v^2 = v_0^2 + 2a(\Delta x)$
\end{definition}
